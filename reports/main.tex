\chapter{GIỚI THIỆU}
\section{Lý do thực hiện đề tài}
Trong bối cảnh nền kinh tế số phát triển mạnh mẽ, các cơ sở tài chính và ngân hàng đang chuyển dịch từ phương thức thẩm định truyền thống sang hệ thống thẩm định tự động dựa trên dữ liệu lớn (Big Data). Rủi ro tín dụng, cụ thể là khả năng khách hàng không thể hoàn trả nợ đúng hạn, là yếu tố sống còn ảnh hưởng trực tiếp đến tỷ lệ nợ xấu (Non-Performing Loans - NPL) và thanh khoản của hệ thống.

Việc xây dựng một mô hình dự báo chính xác không chỉ giúp giảm thiểu rủi ro mà còn cho phép ngân hàng mở rộng tệp khách hàng tiềm năng thông qua việc cá nhân hóa hạn mức tín dụng. Đồ án này tập trung vào bài toán phân loại nhị phân nhằm dự báo xác suất vỡ nợ của khách hàng thẻ tín dụng trong tháng tiếp theo, sử dụng các thuật toán học máy tiên tiến thuộc nhóm Ensemble Learning.

\section{Mục tiêu nghiên cứu}
\begin{itemize}
    \item Phân tích các yếu tố hành vi (lịch sử thanh toán, dư nợ hóa đơn) ảnh hưởng đến khả năng vỡ nợ.
    \item Thực hiện kỹ nghệ đặc trưng (Feature Engineering) để trích xuất các chỉ số tài chính định lượng có ý nghĩa.
    \item Giải quyết vấn đề mất cân bằng dữ liệu - một thách thức phổ biến trong dữ liệu rủi ro tài chính.
    \item So sánh hiệu năng giữa hai mô hình Gradient Boosting phổ biến là LightGBM và XGBoost, đồng thời tối ưu hóa chúng bằng Optuna.
\end{itemize}

\chapter{TỔNG QUAN DỮ LIỆU VÀ PHÂN TÍCH KHÁM PHÁ (EDA)}
\section{Mô tả tập dữ liệu}
Nghiên cứu sử dụng tập dữ liệu \textit{Default of Credit Card Clients} từ Đài Loan. Tập dữ liệu bao gồm 30.000 mẫu với 24 biến độc lập. Theo nghiên cứu gốc của Yeh \& Lien (2009), các biến này bao gồm thông tin cá nhân và dữ liệu tài chính trong vòng 6 tháng từ tháng 4 đến tháng 9 năm 2005.

\section{Phân tích phân phối và Tương quan}
Thông qua quá trình EDA, chúng tôi ghi nhận một số đặc điểm quan trọng từ 30.000 bản ghi dữ liệu:
\begin{itemize}
    \item \textbf{Mất cân bằng lớp (Class Imbalance):} Tỷ lệ khách hàng vỡ nợ (Target = 1) chỉ chiếm khoảng 22.12\% (6.636 mẫu), trong khi khách hàng không vỡ nợ chiếm 77.88\% (23.364 mẫu). Sự lệch pha này yêu cầu phải có các biện pháp xử lý đặc biệt như SMOTE khi huấn luyện mô hình để tránh hiện tượng mô hình bị thiên kiến (bias) về lớp đa số.
    \item \textbf{Đặc điểm nhân khẩu học:} Nhóm khách hàng có độ tuổi từ 25 đến 40 chiếm đa số. Tỷ lệ nữ giới cao hơn nam giới, tuy nhiên tỷ lệ vỡ nợ ở nam giới có xu hướng cao hơn.
    \item \textbf{Đa cộng tuyến (Multicollinearity):} Các biến dự nợ hóa đơn (\texttt{BILL\_AMT1} đến \texttt{BILL\_AMT6}) có hệ số tương quan rất cao (thường > 0.9). Điều này gợi ý rằng các giá trị dư nợ có tính chất lũy kế và ít biến động đột ngột theo tháng đơn lẻ.
    \item \textbf{Trạng thái thanh toán (Payment Status):} Các biến \texttt{PAY\_0} đến \texttt{PAY\_6} có mối tương quan thuận mạnh nhất với biến mục tiêu. Đặc biệt, \texttt{PAY\_0} (trạng thái tháng gần nhất) là biến dự báo đơn lẻ mạnh nhất.
\end{itemize}

\chapter{PHƯƠNG PHÁP NGHIÊN CỨU}
\section{Tiền xử lý dữ liệu và Làm sạch}
Dữ liệu thô thường chứa các giá trị nhiễu hoặc mã hóa không nhất quán. Quy trình làm sạch được thực hiện nghiêm ngặt để đảm bảo tính ổn định:
\begin{itemize}
    \item \textbf{Xử lý biến định danh:} Các nhóm học vấn (\texttt{EDUCATION}) và tình trạng hôn nhân (\texttt{MARRIAGE}) được gộp vào nhóm chính quy hoặc nhóm "Khác" để tránh loãng dữ liệu.
    \item \textbf{Xử lý giá trị âm trong PAY\_x:} Dữ liệu ban đầu chứa các giá trị -1, -2 cho trạng thái thanh toán đúng hạn. Chúng tôi đưa các giá trị này về 0 để mô hình tập trung học các mức độ quá hạn (1, 2, 3...).
\end{itemize}

\section{Kỹ nghệ đặc trưng chuyên sâu (Feature Engineering)}
Dựa trên luận điểm của các hệ thống chấm điểm tín dụng hiện đại, chúng tôi đề xuất trích xuất các biến phi tuyến để tăng cường khả năng phân tách của mô hình:
\begin{enumerate}
    \item \textbf{Tỷ lệ sử dụng tín dụng (\texttt{credit\_util\_ratio}):} 
    \begin{equation}
        \text{Credit\_Util\_Ratio} = \frac{\text{BILL\_AMT1}}{\text{LIMIT\_BAL}}
    \end{equation}
    Biến này phản ánh mức độ "cận biên" của khách hàng. Khách hàng sử dụng sát hạn mức thường có rủi ro cao hơn.
    
    \item \textbf{Tỷ lệ trả nợ (\texttt{payment\_ratio}):}
    \begin{equation}
        \text{Payment\_Ratio}_t = \frac{\text{PAY\_AMT}_{t}}{\text{BILL\_AMT}_{t+1}}
    \end{equation}
    So sánh số tiền trả với dư nợ hóa đơn kỳ trước để xác định hành vi chi trả thực tế.
    
    \item \textbf{Tần suất chậm nợ (\texttt{late\_payment\_count}):} 
    \begin{equation}
        \text{Late\_Count} = \sum_{i \in \{0,2,3,4,5,6\}} [\text{PAY\_i} > 0]
    \end{equation}
    Đếm số lượng trường lịch sử trạng thái thanh toán (\texttt{PAY\_0}, \texttt{PAY\_2} đến \texttt{PAY\_6}) mà khách hàng có trạng thái chậm thanh toán.
\end{enumerate}

\section{Giải quyết mất cân bằng bằng SMOTE}
Kỹ thuật SMOTE (Synthetic Minority Over-sampling Technique) được lựa chọn để tạo mẫu ảo cho lớp thiểu số (Target=1) giúp mô hình học được ranh giới quyết định của nhóm khách hàng rủi ro. Chúng tôi chỉ áp dụng SMOTE trên tập huấn luyện (\textit{Training Set}) để đảm bảo tính khách quan cho tập kiểm thử.

\chapter{KẾT QUẢ VÀ THẢO LUẬN}
\section{Mô hình Boosting và Tối ưu hóa Optuna}
LightGBM và XGBoost được lựa chọn vì khả năng xử lý tốt dữ liệu dạng bảng và các mối quan hệ phi tuyến. 
\begin{itemize}
    \item \textbf{LightGBM} nổi bật với tốc độ xử lý nhanh nhờ kỹ thuật GOSS (Gradient-based One-Side Sampling).
    \item \textbf{XGBoost} được tin dùng nhờ hệ thống chính quy hóa tích hợp sẵn (L1, L2 regularization) giúp kiểm soát phương sai của mô hình.
\end{itemize}

Chúng tôi sử dụng \textbf{Optuna} để thực hiện Bayesian Optimization. Thay vì thử nghiệm ngẫu nhiên, Optuna dựa trên kết quả của các lần thử trước (Trials) để thu hẹp không gian tìm kiếm vào các vùng triển vọng, giúp tìm được bộ siêu tham số tốt nhất với số lần thử nghiệm tối thiểu.

\section{Đánh giá và So sánh hiệu năng}
Kết quả thực nghiệm cho thấy sự vượt trội rõ rệt của các mô hình Boosting so với Baseline Logistic Regression. Đặc biệt, việc tối ưu hóa siêu tham số bằng Optuna đã giúp cải thiện ROC-AUC của cả XGBoost và LightGBM thêm khoảng 0.5-1.0\%.

\subsection{Nhận xét về kết quả}
\begin{itemize}
    \item \textbf{LightGBM (Tuned)} đạt điểm ROC-AUC cao nhất: \textbf{0.7606}, vượt trội so với baseline 1.30 điểm phần trăm (0.7476 → 0.7606) và so với mô hình Default 0.60 điểm phần trăm (0.7546 → 0.7606).
    \item \textbf{Recall của XGBoost (Tuned)} là cao nhất (0.5704), nghĩa là mô hình này bắt được 57\% số lượng khách hàng thực tế sẽ vỡ nợ. Điều này rất quan trọng vì trong ngân hàng, False Negative (bỏ sót khách hàng nguy hiểm) có chi phí cao hơn False Positive.
    \item \textbf{Precision của LightGBM (Tuned)} cao hơn (0.4816), giảm thiểu được sai sót loại I (dự báo sai người không vỡ nợ).
    \item Sự cải thiện sau Optuna tuning (+0.5\% ROC-AUC) cho thấy hiệu quả của phương pháp tìm kiếm siêu tham số thông minh.
\end{itemize}

\section{Ma trận nhầm lẫn (Confusion Matrix)}
Ma trận nhầm lẫn cung cấp cái nhìn chi tiết về 4 loại quyết định của mô hình:
\begin{itemize}
    \item \textbf{TP (True Positive):} Dự báo vỡ nợ + Thực tế vỡ nợ. Số lượng cao tốt.
    \item \textbf{TN (True Negative):} Dự báo không vỡ nợ + Thực tế không vỡ nợ. Số lượng cao tốt.
    \item \textbf{FP (False Positive):} Dự báo vỡ nợ + Thực tế không vỡ nợ. Gây mất khách hàng tiềm năng.
    \item \textbf{FN (False Negative):} Dự báo không vỡ nợ + Thực tế vỡ nợ. **Chi phí cao nhất** - Ngân hàng mất tiền.
\end{itemize}

Với LightGBM (Tuned) có Recall = 0.5433, điều này nghĩa là:
\begin{equation}
    \text{Recall} = \frac{TP}{TP + FN} = 0.5433
\end{equation}

Khoảng 54\% khách hàng quỵt nợ thực tế được phát hiện sớm, giúp ngân hàng triển khai biện pháp phòng chống. Các mô hình đã cải thiện đáng kể so với Baseline (Recall = 0.3534).
